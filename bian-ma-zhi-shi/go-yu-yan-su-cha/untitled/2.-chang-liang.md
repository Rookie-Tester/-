# 2.常量

### 示例代码如下

```text
package main

import "fmt"

func main(){
	//声明方式1
	const a int = 1
	//声明方式2
	const (
		b = 2
		c = 3
	)
	//声明方式3，连续常量
	const (
		d = 1 + iota 	//1
		e				      //2
		f				      //3
		g				      //4
	)
	fmt.Println(d,e,f,g)
	//声明方式4，连续位常量
	const (
		open = 1 << iota	//0001
		close				      //0010
		pending				    //0100
		success				    //1000
	)
	fmt.Println(open,close,pending,success)
}
```
